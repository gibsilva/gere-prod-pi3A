public void excluirCliente(ClienteBeans Cliente){
        try {
            //linha de exclusão do produto com a conexão com o banco
            String SQLInsertion = "delete from clientes where cli_cod = ? ";
            PreparedStatement st = Conexao.getConnection().prepareStatement(SQLInsertion);
            
            st.setInt(1, Cliente.getCodigo());
            
            //executar o programa
            st.execute();
            Conexao.getConnection().commit();
            JOptionPane.showMessageDialog(null, "Registro excluir com sucesso.", "excluir", 1);
            
        } catch (SQLException ex) {
            //JOptionPane.showMessageDialog(null, ex, "Erro", 0);
            JOptionPane.showMessageDialog(null, "Erro ao excluir registro", "Erro", 0);
        }
    }
